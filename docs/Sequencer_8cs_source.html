174 private void UpdateSequencerStatus()
175 {
176 _songTime = _song.GetSongTime() + _offset + _lag;
177 _statusResolver.UpdateStatus(_songTime);
178 var state = _statusResolver.GetStatus();
179
180 _warpedTime = state.WarpedTime;
181 _delayedTime = state.DelayedTime;
182 _stoppedTime = state.StoppedTime;
183 _beat = state.Beat;
184 _position = state.Position;
185 _speed = state.Speed;
186 _combos = state.Combos;
187
188 _speed *= _userSpeed;
189 _percentageOfBeat = 1 - (_beat % 1);
190 _state = state;
191 }
