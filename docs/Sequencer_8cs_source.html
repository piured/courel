119 public void LoadChart(IChart loader)
120 {
121 _statusResolver = new StateResolver(loader);
122 _composer = new Composer(loader, _statusResolver);
123 _runtimeResolver = new RunTimeResolver(
124 _judge,
125 _composer.GetSingleNoteScore(),

133 void Update()

137 UpdateSequencerStatus();

139 }

142 private void UpdateSequencerStatus()
143 {
144 _songTime = _song.GetSongTime() + _offset + _lag;
145 _statusResolver.UpdateStatus(_songTime);
146 var state = _statusResolver.GetStatus();
147
148 _warpedTime = state.WarpedTime;
149 _delayedTime = state.DelayedTime;
150 _stoppedTime = state.StoppedTime;
151 _beat = state.Beat;
152 _position = state.Position;
153 _speed = state.Speed;
154 _combos = state.Combos;
155
156 _speed *= _userSpeed;
157 _percentageOfBeat = 1 - (_beat % 1);
158 _state = state;
159 }
