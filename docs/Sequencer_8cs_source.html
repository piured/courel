180 void Update()
181 {
182 if (_statusResolver != null && _runtimeResolver != null)

184 UpdateSequencerStatus();

189 private void UpdateSequencerStatus()
190 {
191 _songTime = _song.GetSongTime() + _offset + _userOffset + _inputLag;
192 _songTimeWithoutInputLag = _song.GetSongTime() + _offset + _userOffset;
193 _statusResolver.UpdateStatus(_songTime);
194 var state = _statusResolver.GetStatus();
195
196 _warpedTime = state.WarpedTime;
197 _delayedTime = state.DelayedTime;
198 _stoppedTime = state.StoppedTime;
199 _beat = state.Beat;
200 _position = state.Position;
201 _speed = state.Speed;
202 _combos = state.Combos;
203
204 _speed *= _userSpeed;
205 _percentageOfBeat = 1 - (_beat % 1);
206 _state = state;
207 }
